@列出所有的数据库

mysql> show databases;
+--------------------+
| Database           |
+--------------------+
| information_schema |
| java               |
| kisscss_wp         |
| mysql              |
| performance_schema |
| test               |
+--------------------+

@创建数据库

mysql> create database php;

@删除数据库

mysql> drop database php;

@重命名数据库

@选择数据库

mysql> use php;

@备份数据库

注(mysqldump的路径)：/opt/mysql/bin/mysqldump

mysqldump DBname > backupDBname.sql

@导出结构不导出数据

mysqldump -d 数据库名 -uroot -p > xxx.sql

@导出数据不导出结构

mysqldump -t 数据库名 -uroot -p > xxx.sql

@导出数据和表结构

mysqldump 数据库名 -uroot -p > xxx.sql

@导出特定表的结构

mysqldump -uroot -p -B 数据库名 --table 表名 > xxx.sql

导出整个数据库结构和数据

mysqldump -h localhost -uroot -p123456 database > dump.sql

导出单个数据表结构和数据

mysqldump -h localhost -uroot -p123456  database table > dump.sql

导出整个数据库结构（不包含数据）

mysqldump -h localhost -uroot -p123456  -d database > dump.sql

导出单个数据表结构（不包含数据）

mysqldump -h localhost -uroot -p123456  -d database table > dump.sql

@导入数据库

mysql>use 目标数据库名
mysql>source 导入的文件名
或者：
/opt/mysq/bin/mysql DBname < fileName.sql

注：导出的数据库文件在shell当前路径下(pwd)

@修改数据库密码

mysql>use mysql
mysql>UPDATE user SET password=PASSWORD("new password") WHERE user='name';
mysql> FLUSH PRIVILEGES;

@新建表

mysql> create table users (id int(6),name VARCHAR(20));

@新建索引

mysql> create index users_name on users (name);

@列出所有的表

mysql> show tables;
+---------------+
| Tables_in_php |
+---------------+
| user          |
+---------------+

@修改auto_increment

mysql> alter table users AUTO_INCREMENT=2;
+---------------+
| Tables_in_php |
+---------------+
| user          |
+---------------+

mysql> select sid,ip,FROM_UNIXTIME(create_time,'%Y-%m-%d %H:%i:%s') from yy_getcode_session;

@删除唯一键

mysql> alter table user drop index name;

@删除列

mysql> alter table user drop column age;

@修改列数据类型和编码

mysql> alter table user modify column name varchar(33) character set utf8;

@将id字段设为主键并自增

mysql> alter table user change id id int primary key auto_increment;

或者

mysql> alter table user modify column id int(4) primary key auto_increment;
注:以上条sql语句的运行条件是id不是主键。

@删除user表的主键属性

mysql> alter table user drop primary key,change id id int(5);

@插入数据

mysql> insert into user values(1,'zhangheng');
mysql> insert into user values(2,'zhangzhang'),(3,'hengzhang'),(4,'hengheng');

@清空数据表

delete from tablename
truncate table tablename

@更新数据表

mysql> update user set name='kisscss' where id=1;

@给远程访问客户端权限

grant all on *.* to 'username'@'192.168.0.233' identified by 'password'
grant all on *.* to duoli1003a@'192.168.1.%' identified by 'fqd^ui@24wx*'

username:表示用户名.
192.168.0.233:远程ip地址
password:表示远程登陆密码.
然后在233那太电脑上就可以如下方式访问mysql:

mysql -h 192.168.50.108 -u username -p

@显示某个数据库所有的存储过程名

show procedure status where db='dbname'

@显示存储过程内容

show create procedure proName

@在命令行创建存储过程

注：在命令行一般遇到；就表示语句结束了，所以可以通过改变结束符（delimiter）来避免这个冲突
delimiter //
这样就改变了结束符号为//，可以做如下测试：

mysql> delimiter //
mysql> select * from user;
    -> //
+------+------------+
| id   | name       |
+------+------------+
|    1 | zhangheng  |
|    2 | zhangzhang |
|    3 | hengzhang  |
|    4 | hengheng   |
+------+------------+

说明结束符已经替换成了//，如果想换回来，可以这样：
delimiter ;

@创建存储过程

mysql> create procedure chaxun()
    -> begin
    -> select * from user;
    -> end;
    -> //

@调用存储过程

mysql> call chaxun();
+------+------------+
| id   | name       |
+------+------------+
|    1 | kisscss    |
|    2 | zhangzhang |
|    3 | hengzhang  |
|    4 | hengheng   |
+------+------------+

@删除存储过程

mysql> drop procedure pr_empty;
